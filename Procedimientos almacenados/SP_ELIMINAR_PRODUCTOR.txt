CREATE OR REPLACE PROCEDURE SP_ELIMINAR_PRODUCTOR (
INT_ID_PRODUCTOR IN NUMBER,
OUT_GLOSA 	OUT VARCHAR2,
OUT_ESTADO 	OUT NUMBER
)
AS


BEGIN

 UPDATE FERIA2.PRODUCTOR SET habilitado=0 WHERE ID=INT_ID_PRODUCTOR ;

 EXCEPTION
        WHEN OTHERS THEN
            OUT_ESTADO := -1;
            OUT_GLOSA := feria2.FN_GET_GLOSA_ERROR;
END SP_ELIMINAR_PRODUCTOR;
